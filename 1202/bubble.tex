
\documentclass{article}
\usepackage{tikz}
\begin{document}
    \begin{tikzpicture}
        \pgfmathsetseed{20251202}
        \foreach \x in {1, ..., 10} {
            \foreach \y in {1, ..., 15} {
                \pgfmathsetmacro\r{rnd} % random number in [0, 1]
                \pgfmathsetmacro\u{rnd/2} % random number in [0, 1/2]
                \pgfmathsetmacro\v{rand/3} % random number in [-1/3, 1/3]
                \pgfmathsetmacro\c{rnd*100} % random number in [0, 100]
                \draw [red!\c!blue] (\x + \u, \y + \v)
                    circle [x radius=\r, y radius=1-\r];
            }
        }
    \end{tikzpicture}
\end{document}
